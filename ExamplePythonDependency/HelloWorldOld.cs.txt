        {
            // TODO : See PythonEnvironment for actions on venv's and paths.

            var env = new PythonEnvironment("C:\\Users\\anthonyshaw\\projects\\Build2024AspireDemo\\ExamplePythonDependency\\");

            {
                // convert the Person object to a PyObject
